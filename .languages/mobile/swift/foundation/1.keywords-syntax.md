# Swift Keywords and Syntax for Mobile Development

## Swift Keywords (Complete List)

### Declaration Keywords
```swift
class       // Class declaration (UIViewController, custom views)
struct      // Value type (SwiftUI Views, model objects)
enum        // Enumeration (app states, error types)
protocol    // Protocol declaration (delegates, data sources)
extension   // Extend existing types
func        // Function declaration
var         // Mutable variable
let         // Immutable constant
typealias   // Type alias
associatedtype  // Associated type in protocols
```

### Access Control Keywords
```swift
open        // Most permissive (cross-module inheritance)
public      // Cross-module access (framework APIs)
internal    // Module-internal (default, app-wide)
fileprivate // File-private access
private     // Enclosing declaration only
```

### Control Flow Keywords
```swift
if          // Conditional statement
else        // Alternative branch
switch      // Multi-way branch (enum matching)
case        // Switch case
default     // Default switch case
for         // Loop iteration
while       // While loop
repeat      // Repeat-while loop
break       // Exit loop/switch
continue    // Skip iteration
return      // Return from function
guard       // Early exit with unwrapping
where       // Additional conditions
```

### Error Handling Keywords
```swift
do          // Error handling block
try         // Attempt throwing operation
catch       // Handle errors
throw       // Throw error
throws      // Function can throw
rethrows    // Function rethrows from parameters
defer       // Cleanup code execution
```

### Memory Management Keywords
```swift
weak        // Weak reference (preventing retain cycles)
unowned     // Unowned reference (assume valid)
strong      // Strong reference (default)
```

### Async/Await Keywords (iOS 13+)
```swift
async       // Asynchronous function
await       // Wait for async operation
actor       // Actor type for concurrency
```

### Property Keywords
```swift
get         // Property getter
set         // Property setter
willSet     // Property observer (before set)
didSet      // Property observer (after set)
lazy        // Lazy initialization
```

### Inheritance Keywords
```swift
class       // Reference type with inheritance
final       // Prevent inheritance/override
override    // Override superclass member
super       // Reference to superclass
init        // Initializer
deinit      // Deinitializer
required    // Required initializer
convenience // Convenience initializer
```

### Generic Keywords
```swift
where       // Generic constraints
some        // Opaque return types
Any         // Any type
AnyObject   // Any class type
```

### Literal Keywords
```swift
true        // Boolean true
false       // Boolean false
nil         // Null value
self        // Current instance
Self        // Current type
```

### Advanced Keywords

## Swift Syntax for Mobile Development

### Basic Variable Declaration
```swift
// Constants (preferred for immutable data)
let appName = "MyApp"
let version: String = "1.0.0"
let isProVersion: Bool = true

// Variables (for mutable state)
var userScore = 0
var selectedItems: [String] = []
var currentViewController: UIViewController?

// Lazy properties (for expensive initialization)
lazy var locationManager = CLLocationManager()
lazy var imageCache = NSCache<NSString, UIImage>()
```

### Optionals (Critical for iOS Development)
```swift
// Optional declaration
var username: String? = nil
var profileImage: UIImage?

// Optional binding (safe unwrapping)
if let user = currentUser {
    print("Welcome, \(user.name)")
    updateUI(for: user)
}

// Guard statements for early exit
guard let data = response.data else {
    showError("No data received")
    return
}

// Nil coalescing operator
let displayName = user.name ?? "Unknown User"
let avatarImage = user.avatar ?? UIImage(named: "default-avatar")

// Optional chaining
let streetName = user.address?.street?.name
navigationController?.pushViewController(vc, animated: true)

// Force unwrapping (use sparingly, when certain)
let image = UIImage(named: "logo")!
```

### Functions and Closures
```swift
// Basic function
func calculateTotal(price: Double, tax: Double) -> Double {
    return price + (price * tax)
}
```

### Classes and Inheritance (UIKit Pattern)
```swift
// UIViewController subclass
class HomeViewController: UIViewController {
    // IBOutlet connections
    @IBOutlet weak var titleLabel: UILabel!
    @IBOutlet weak var tableView: UITableView!
    
    // Properties
    private let viewModel = HomeViewModel()
    private var dataSource: [Item] = []
    
    // Lifecycle methods
    override func viewDidLoad() {
        super.viewDidLoad()
        setupUI()
        bindViewModel()
    }
    
    override func viewWillAppear(_ animated: Bool) {
        super.viewWillAppear(animated)
        refreshData()
    }
    
    // IBAction methods
    @IBAction func refreshButtonTapped(_ sender: UIButton) {
        refreshData()
    }
    
    // Private methods
    private func setupUI() {
        title = "Home"
        tableView.delegate = self
        tableView.dataSource = self
    }
}

// Protocol conformance extension
extension HomeViewController: UITableViewDataSource {
    func tableView(_ tableView: UITableView, numberOfRowsInSection section: Int) -> Int {
        return dataSource.count
    }
    
    func tableView(_ tableView: UITableView, cellForRowAt indexPath: IndexPath) -> UITableViewCell {
        let cell = tableView.dequeueReusableCell(withIdentifier: "ItemCell", for: indexPath)
        configure(cell: cell, with: dataSource[indexPath.row])
        return cell
    }
}
```

### Enumerations (State Management)
```swift
// App state enum
enum AppState {
    case loading
    case loaded(data: [Item])
    case error(message: String)
    case empty
}

// Network result enum
enum NetworkResult<T> {
    case success(T)
    case failure(NetworkError)
}

// Enum with associated values
enum NetworkError: LocalizedError {
    case noConnection
    case timeout
    case serverError(code: Int)
    case invalidData
    
    var errorDescription: String? {
        switch self {
        case .noConnection:
            return "No internet connection"
        case .timeout:
            return "Request timed out"
        case .serverError(let code):
            return "Server error: \(code)"
        case .invalidData:
            return "Invalid data received"
        }
    }
}

// Using enums with switch
func handleAppState(_ state: AppState) {
    switch state {
    case .loading:
        showLoadingSpinner()
    case .loaded(let data):
        updateUI(with: data)
    case .error(let message):
        showError(message)
    case .empty:
        showEmptyState()
    }
}
```

### Async/Await (iOS 13+)
```swift
// Async function
func fetchUserProfile(for userID: String) async throws -> User {
    let url = URL(string: "https://api.example.com/users/\(userID)")!
    let (data, _) = try await URLSession.shared.data(from: url)
    return try JSONDecoder().decode(User.self, from: data)
}

// Using async/await in SwiftUI
struct ProfileView: View {
    @State private var user: User?
    @State private var isLoading = false
    
    var body: some View {
        Group {
            if let user = user {
                UserDetailView(user: user)
            } else if isLoading {
                ProgressView("Loading...")
            } else {
                Text("No user data")
            }
        }
        .task {
            await loadUser()
        }
    }
    
    private func loadUser() async {
        isLoading = true
        defer { isLoading = false }
        
        do {
            user = try await fetchUserProfile(for: "123")
        } catch {
            print("Failed to load user: \(error)")
        }
    }
}
```

### Error Handling Patterns
```swift
// Do-try-catch
func saveUserData(_ data: Data) {
    do {
        let user = try JSONDecoder().decode(User.self, from: data)
        try CoreDataManager.shared.save(user)
        showSuccessMessage()
    } catch DecodingError.dataCorrupted {
        showError("Invalid user data format")
    } catch CoreDataError.saveFailed {
        showError("Failed to save user data")
    } catch {
        showError("An unexpected error occurred: \(error.localizedDescription)")
    }
}

// Result type pattern
func loadImage(from url: URL, completion: @escaping (Result<UIImage, Error>) -> Void) {
    URLSession.shared.dataTask(with: url) { data, response, error in
        if let error = error {
            completion(.failure(error))
            return
        }
        
        guard let data = data, let image = UIImage(data: data) else {
            completion(.failure(ImageError.invalidData))
            return
        }
        
        completion(.success(image))
    }.resume()
}
```

### Combine Framework Integration
```swift
import Combine

class ViewModel: ObservableObject {
    @Published var items: [Item] = []
    @Published var isLoading = false
    @Published var searchText = ""
    
    private var cancellables = Set<AnyCancellable>()
    private let apiService = APIService()
    
    init() {
        // Search with debounce
        $searchText
            .debounce(for: .milliseconds(300), scheduler: RunLoop.main)
            .removeDuplicates()
            .sink { [weak self] searchText in
                self?.performSearch(query: searchText)
            }
            .store(in: &cancellables)
    }
    
    func performSearch(query: String) {
        isLoading = true
        
        apiService.search(query: query)
            .receive(on: DispatchQueue.main)
            .sink(
                receiveCompletion: { [weak self] completion in
                    self?.isLoading = false
                    if case .failure(let error) = completion {
                        print("Search failed: \(error)")
                    }
                },
                receiveValue: { [weak self] items in
                    self?.items = items
                }
            )
            .store(in: &cancellables)
    }
}
```

### Memory Management Patterns
```swift
// Avoiding retain cycles in closures
class NetworkManager {
    var completion: ((Data?) -> Void)?
    
    func fetchData() {
        URLSession.shared.dataTask(with: url) { [weak self] data, _, _ in
            // Use weak self to avoid retain cycle
            self?.completion?(data)
        }.resume()
    }
}

// Weak delegate pattern
protocol ViewControllerDelegate: AnyObject {
    func didComplete()
}

class ViewController: UIViewController {
    weak var delegate: ViewControllerDelegate?
    
    @IBAction func finishButtonTapped() {
        delegate?.didComplete()
    }
}
```
